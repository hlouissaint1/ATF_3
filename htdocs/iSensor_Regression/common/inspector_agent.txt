*** Settings ***
Documentation     Commmon utilities to handle/validate iSensor Inspector Agents

Library		OperatingSystem
Library		SSHLibrary
Library		BuiltIn

*** Keywords ***
Disable Inspector Service
    FOR     ${setting}    IN RANGE     10
        ${inspinit_status}=    Execute Command    svstat /service/inspinit/ | awk '{print $2}'
        Exit For Loop If    'down'=='${inspinit_status}'
	BuiltIn.Run Keyword If    'up' == '${inspinit_status}'    Stop Inspinit Service
	BuiltIn.Sleep    2
    END


Enable Inspector Service
    FOR     ${setting}    IN RANGE     10
        ${inspinit_status}=    Execute Command    svstat /service/inspinit/ | awk '{print $2}'
        Exit For Loop If    'up'=='${inspinit_status}'
	BuiltIn.Run Keyword If    'down' == '${inspinit_status}'    Start Inspinit Service
        BuiltIn.Sleep    2
    END


Stop Inspinit Service
    write    svc -d /service/inspinit


Start Inspinit Service
    write    svc -u /service/inspinit
